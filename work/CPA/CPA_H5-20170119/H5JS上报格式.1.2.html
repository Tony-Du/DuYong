pageVisit的格式
页面访问时长

{

measure:{

       "URL":"http://127.0.0.1/migu/aa.jsp",

       "OS":"iOS",

       "timeStamp":"1480521763049"

},

param: {

          "type":"pageVisit",

          "cookieId":"asdf435234fqw2323vwe34",

          "IP":"127.0.0.1",

          "startTime":"1480521763049", //页面打开时间

          "endTime":"1480521763049", //页面退出时间

          "visitTime": endTime - startTime //差值
       }
}

logEvent的格式:

migu.logEvent(downloadClick,{'description':'xxxxxx','aaa':'bbb'}); 目前用来计算用于下载点击次数，以后可以扩展

{

measure:{

        "URL":"http://127.0.0.1/migu/aa.jsp",

        "OS":"iOS",

        "timeStamp":"1480521763049"

},

param: {

            "type":"downloadClick",

            "description":"xxxxxx",

            "aaa":"bbb"  // 扩展字段

        }
}
